1. Find the Smaller Angle.

    var Minimal_Angle = (h, m) => {
    let hAngle=((h*360)/12+(m*360)/(12*60))
    let mAngle=((m*360)/60)
    let angle=Math.abs(hAngle-mAngle)
    if(angle>180){
        angle=360-angle
    }
    return angle
    };

3. Perfect Number Check.

    var Perfect_Check = (N) => {
    sum = 1;
    if(N===1){
        return "YES"
    }
    else if(N%2===1){
        return "NO"
    }
    else{
        
        for(let i=2;i*i<=N;i++){
        if(N%i===0){
            sum += i+N/i
        }
        }
        
        return sum===N?"YES":"NO"
    }
    };

4. Reverse a Number.

    var Reverse_Number = (N) => {
    N=N+""
    let rev = N.split("").reverse().join("")
    return parseInt(rev)
    };
